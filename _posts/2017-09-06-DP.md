public int maxA(int N) {
  int[] dp = new int[N+1];
  for (int i = 0; i < N+1; i++) {
    dp[i] = i;
    for (int j = 3; j < i; j++) {
      dp[i] = Math.max(dp[i], dp[j-i] * (j-1));
    }
  }
  return dp[N];
}
